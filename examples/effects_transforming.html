transforming

rotate model

  • setRotate("saw", [0,0,0]); // Rotate saw 0 degrees about X
  • setRotate("saw", [90,0,0]); // Rotate saw 90 degrees about X

translate model

  • setTranslate("saw", [0,0,-100]); // Translate saw backwards along -Z
  • setTranslate("saw", [0,0,0]); // Restore saw translation to default

scale model

  • setScale("saw", [0.5,0.5,0.5]); // Scale saw by 50%
  • setScale("saw", [1,1,1]); // Restore saw scale to default
